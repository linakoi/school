;アセンブリ言語命令
;ソフトウェアタイマー
	.CPU	300HA
	.SECTION 	PROG,DATA,LOCATE=H'00000
	.DATA.L	RESET
	.SECtION	P,CODE,LOCATE=H'00800
RESET:	MOV.L	#H'FFFF00,ER7	;SPセット
		MOV.B	#H'FF,R0L
		MOV.B	R0L,@H'EE00A	;PB = 出力
		MOV.B	#H'FF,R0L
		MOV.B	R0L,@H'FFFDA	;H'FF出力
		MOV.L	#D'3125000,ER6	;	1秒タイマーセット
TIMER:	NOP;時間調整
		DEC.L	#1,ER6	;ER6を1減らす
		BNE		TIMER	;ER6 !=	0
		MOV.B	#H'00,R0L
		MOB.V	R0L,@H'FFFDA	;@H'00
LOOP:	NOP	;無限ループ
		JMP		@LOOP
		.END
