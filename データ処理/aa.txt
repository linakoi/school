int main () {

            n = i-1;
            break;
        }
    }

    for(int i = 0;i < 726;i++) {
        printf("\n相手HP%d\n",chp);
        printf("自分HP%d\n",php);
        printf("カウンター：0　打撃：1 魔法:2\n");

                flag *= -1;
            }
        }
        if(c == 2) {

                printf("相手にダメージ！\n");
            } else {
                printf("自分にダメージ！\n");
                flag *= -1;
            }
        }

            printf("クリティカルヒット！");
            flag*=2;
        }
        if(flag > 0) {

        } else {
            php += flag;
        }
    }
}
